                    let h_bie;
                    do {
                        var liczba1 = prompt("Podaj wysokość trójkąta:", "");
                        h_bie = parseFloat(liczba1);
                    } while (h_bie <= 0);

                    let a_bie;
                    do {
                        var liczba2 = prompt("Podaj podstawę trójkąta:", "");
                        a_bie = parseFloat(liczba2);
                    } while (!(a_bie > 0));

                    var Pole = a_bie * h_bie / 2;
                    document.write("Pole trójkąta: " + Pole.toFixed(2) + " cm<sup>2</sup><br>");
